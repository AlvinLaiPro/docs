## BSS add language key 
`set BSS_USE_DEFAULT_PROPAGATION=true && node --experimental-modules ./scripts/propagateKeys.js --valuePath=values.json`
```javascript
import fse from 'fs-extra';
import yargs from 'yargs'
import { hideBin } from 'yargs/helpers'
import {join} from 'path';

const {readdir, readJSON, writeFile} = fse;
const argv = yargs(hideBin(process.argv))
    .options({
        'valuePath': {
            type: 'string',
            description: 'value json file path',
        }
    })
    .argv;

(async () => {
    try {
        const valuePath = argv.valuePath;
        await main(valuePath);
    } catch (err) {
        console.error(err);
        process.exit(1);
    }
})();

async function main(valuePath) {
    const srcDirs = (await readdir('./src', {withFileTypes: true}))
        .filter((dirent) => dirent.isDirectory())
        .map((dirent) => dirent.name);
    const valueJson = valuePath ? await readJSON(join(process.cwd(), valuePath)) : null;

    Promise.all(
        srcDirs.map(async (assetDir) => {
            console.log('debug ~ file: propagateKeys.js:21 ~ srcDirs.map ~ assetDir:', assetDir);
            const bssUseDefaultPropagation = process.env.BSS_USE_DEFAULT_PROPAGATION?.trim();
            const enUSLocaleSource = await readJSON(`./src/${assetDir}/en-US.json`);
            const enUSKeys = Object.keys(enUSLocaleSource);
            let shouldSkipEnUS = true;
            let targetKey = '';

            if (valueJson && valueJson[assetDir]) {
                targetKey = Object.keys(valueJson[assetDir]['en-US'])[0];

                if (!enUSKeys.includes(targetKey)) {
                    enUSKeys.push(targetKey);
                    shouldSkipEnUS = false;
                }
            }

            const enUSThemesDict = enUSLocaleSource.themes.reduce((acc, next) => {
                acc[next.themeId] = next;
                return acc;
            }, {});
            const enUSThemesCount = Object.keys(enUSThemesDict).length;

            const otherLocaleFilesList = (await readdir(`./src/${assetDir}`, {withFileTypes: true}))
                .filter(
                    (dirent) =>
                        !dirent.isDirectory() &&
                        /^\w\w-[\w\d]{2,3}\.json$/gi.test(dirent.name) && (shouldSkipEnUS ? dirent.name !== 'en-US.json' : true),
                )
                .map((dirent) => dirent.name);

            await Promise.all(
                otherLocaleFilesList.map(async (locale) => {
                    const currentLocaleSource = await readJSON(`./src/${assetDir}/${locale}`);
                    enUSKeys.forEach((key) => {
                        if (typeof currentLocaleSource[key] === 'undefined') {

                            if (key === targetKey && valueJson[assetDir]) {
                                const language = locale.replace('.json', '');
                                currentLocaleSource[key] = valueJson[assetDir][language][targetKey];
                            } else {
                                currentLocaleSource[key] = (
                                    bssUseDefaultPropagation === 'true' ||
                                    bssUseDefaultPropagation === 'TRUE'
                                )
                                    ? enUSLocaleSource[key]
                                    : '';
                            }
                        }
                    });

                    const removeThemes = [];

                    currentLocaleSource.themes.forEach((theme) => {
                        if (!enUSThemesDict[theme.themeId]) {
                            removeThemes.push(theme.themeId);
                        } else {
                            Object.entries(enUSThemesDict[theme.themeId]).forEach(([enUSthemeKey, enUSthemeValue]) => {
                                if (typeof theme[enUSthemeKey] === 'undefined') {
                                    theme[enUSthemeKey] = enUSthemeValue;
                                }
                            });
                        }
                    });

                    currentLocaleSource.themes = currentLocaleSource.themes.filter(
                        (theme) => !removeThemes.includes(theme.themeId),
                    );

                    if (currentLocaleSource.themes.length !== enUSThemesCount) {
                        enUSLocaleSource.themes.forEach((enUSTheme) => {
                            if (!currentLocaleSource.themes.find((theme) => theme.themeId === enUSTheme.themeId)) {
                                currentLocaleSource.themes.push(enUSTheme);
                            }
                        });
                    }

                    await writeFile(
                        `./src/${assetDir}/${locale}`,
                        JSON.stringify(currentLocaleSource, null, 4),
                        {
                            encoding: 'utf8',
                        },
                    );
                }),
            );
        }),
    );
}
```
